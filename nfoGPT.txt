using namespace System;

namespace TestKeys
{
    internal static class Keybindings
    {
        private const int ESC = 0x1B;
        private const int F = 0x1C;
        private const int R = 0x1D;
        private const int I = 0x27;

        internal static bool isFKeyDown()
        {
            return GetKeyState(ref F);
        }

        internal static bool isRKeyDown()
        {
            return GetKeyState(ref R);
        }

        internal static bool isIKeyDown()
        {
            return GetKeyState(ref I);
        }

        internal static bool checkKeysCombined(params Keys[] keys)
        {
            bool e = isEscKeyDown();
            bool r = isRKeyDown();
            bool i = isIKeyDown();

            for (int i = 0; i < keys.Length; ++i)
            {
                if ((keys[i].keyCode == ESC && e) ||
                   (keys[i].keyCode == F && r) ||
                   (keys[i].keyCode == I && i))
                {
                    return true;
                }
            }

            return false;
        }

        internal static void printKeybindings()
        {
            Console.WriteLine("ESC: {0x1B}", ES);
            Console.WriteLine("F: {0x1C}", F);
            Console.WriteLine("R: {0x1D}", R);
            Console.WriteLine("I: {0x27}", I);
        }
    };
}
